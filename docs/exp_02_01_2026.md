pythonclass AdaptiveNormalizer:
    """Робастная нормализация с адаптацией к субъекту"""
    
    def __init__(self, method='robust_scale'):
        self.method = method
        self.global_stats = None
        self.subject_stats = {}
    
    def fit(self, segments: dict, subject_id: str):
        # 1. Глобальная статистика (все субъекты)
        all_data = np.concatenate([s for segs in segments.values() 
                                   for s in segs], axis=0)
        
        # Используем Huber estimator (робастнее медианы)
        from sklearn.covariance import MinCovDet
        mcd = MinCovDet().fit(all_data.reshape(-1, all_data.shape[-1]))
        
        self.global_stats = {
            'loc': mcd.location_,      # robust центр
            'scale': np.sqrt(mcd.covariance_.diagonal())  # robust масштаб
        }
        
        # 2. Статистика конкретного субъекта
        self.subject_stats[subject_id] = {
            'loc': np.median(all_data.reshape(-1, all_data.shape[-1]), axis=0),
            'scale': 1.4826 * np.median(
                np.abs(all_data.reshape(-1, all_data.shape[-1]) - 
                       self.global_stats['loc']), axis=0
            )
        }
    
    def transform(self, segment: np.ndarray, 
                  subject_id: str = None,
                  alpha: float = 0.3) -> np.ndarray:
        """
        alpha: вес subject-specific stats (0.3 = 30% subject, 70% global)
        """
        if subject_id and subject_id in self.subject_stats:
            # Смешиваем глобальную и subject-specific статистику
            loc = (1-alpha) * self.global_stats['loc'] + \
                  alpha * self.subject_stats[subject_id]['loc']
            scale = (1-alpha) * self.global_stats['scale'] + \
                    alpha * self.subject_stats[subject_id]['scale']
        else:
            loc = self.global_stats['loc']
            scale = self.global_stats['scale']
        
        normalized = (segment - loc[None, None, :]) / (scale[None, None, :] + 1e-8)
        return np.clip(normalized, -5, 5)  # жесткий клиппинг
